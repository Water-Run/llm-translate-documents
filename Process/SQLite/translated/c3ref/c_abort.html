#define SQLITE_READONLY     8   /* 尝试写入只读数据库 */

#define SQLITE_NOTFOUND    12   /* 在 sqlite3_file_control() 中遇到未知操作码 */

#define SQLITE_CANTOPEN    14   /* 无法打开数据库文件 */
#define SQLITE_PROTOCOL    15   /* 数据库锁协议错误 */

#define SQLITE_SCHEMA      17   /* 数据库模式已更改 */
